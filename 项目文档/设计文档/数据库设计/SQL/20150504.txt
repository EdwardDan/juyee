/*==============================================================*/
/* Table: OA_BBS                                                */
/*==============================================================*/
create table OA_BBS  (
   ID                   NUMBER(10)                      not null,
   TYPE_ID              NUMBER(10),
   TYPE_NAME            VARCHAR2(200 CHAR),
   TITLE                VARCHAR2(200 CHAR),
   CONTENT              CLOB,
   VISIT_TIMES          NUMBER(8)                      default 0,
   DOCUMENT_ID          NUMBER(10),
   PARENT_ID            NUMBER(10),
   CREATE_TIME          DATE,
   CREATE_USER          VARCHAR2(100),
   UPDATE_TIME          DATE,
   UPDATE_USER          VARCHAR2(100),
   PERSON_NAME          VARCHAR2(100),
   IS_VALID             NUMBER(1)                      default 1,
   IS_TOP               NUMBER(1)                      default 0,
   REPLY_TIMES          NUMBER(8)                      default 0,
   DEPT_NAME            VARCHAR2(200 CHAR),
   constraint PK_OA_BBS primary key (ID),
   constraint FK_OA_BBS_PARENT foreign key (PARENT_ID)
         references OA_BBS (ID),
   constraint FK_OA_BBS_TYPE foreign key (TYPE_ID)
         references SYS_CODE_DETAIL (ID),
   constraint FK_OA_BBS_DOC foreign key (DOCUMENT_ID)
         references DOC_DOCUMENT (ID)
);
